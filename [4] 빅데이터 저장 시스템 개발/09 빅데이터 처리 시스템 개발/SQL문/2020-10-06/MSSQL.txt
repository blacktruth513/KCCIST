CREATE database shopping_mall;

DROP TABLE member;
CREATE TABLE member (
	member_id 		CHAR(25) 	NOT NULL,
	member_name 	NCHAR(10) 		NOT NULL,
	member_addr 	NCHAR(255)	NULL,
	made_date		DATETIME		DEFAULT GETDATE() NULL
	
	PRIMARY KEY(member_id)
);

DROP TABLE product;
CREATE TABLE product (
	product_name 	NCHAR(10) 	NOT NULL,
	product_pri 	INT 		NOT NULL, 
	product_day		datetime NOT NULL,
	product_com		NCHAR(10)	NOT NULL,
	product_cnt		INT		NOT NULL, 
	made_date		DATETIME		DEFAULT GETDATE() null
	
 	PRIMARY KEY(product_name)
);

SELECT member_id AS 아이디, member_name AS '회원이름', member_addr AS 주소  
FROM member
ORDER BY made_date ASC;

SELECT product_name AS 제품이름, product_pri AS 제품가격,product_day AS 제조일자, product_com AS 제조회사, product_cnt AS 남은수량
